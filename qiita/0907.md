
 EC風サイトを作る

#フォルダ構成

```
ec-app/
  qiita/
  backend/    ← Laravel を入れる場所
  frontend/   ← Next.js を入れる場所
  docker/
    php/
    node/
    nginx/
      conf.d/

```

ディレクトリ作成
```
ステップ1：ディレクトリ作成

ec-app/ に移動して、以下を実行してください👇

cd ec-app

# backend と frontend のディレクトリを作成
mkdir backend frontend

# docker 用のディレクトリも作成
mkdir -p docker/php docker/node docker/nginx/conf.d
```
"-p"とは？
→親フォルダも作ってくれる

docker-compose.yml を作成
```
version: "3.9"

services:
  app:
    build:
      context: ./docker/php
      dockerfile: Dockerfile
    container_name: laravel_app
    working_dir: /var/www/html
    volumes:
      - ./backend:/var/www/html
    environment:
      - TZ=Asia/Tokyo
    depends_on:
      - db
      - redis
    networks:
      - appnet

  web:
    image: nginx:1.27-alpine
    container_name: laravel_web
    ports:
      - "8080:80"
    volumes:
      - ./backend:/var/www/html
      - ./docker/nginx/conf.d:/etc/nginx/conf.d
    depends_on:
      - app
    networks:
      - appnet

  db:
    image: mysql:8.0
    container_name: laravel_db
    environment:
      MYSQL_DATABASE: app
      MYSQL_USER: app
      MYSQL_PASSWORD: secret
      MYSQL_ROOT_PASSWORD: root
      TZ: Asia/Tokyo
    ports:
      - "3307:3306"
    volumes:
      - dbdata:/var/lib/mysql
    command: [
      "--character-set-server=utf8mb4",
      "--collation-server=utf8mb4_0900_ai_ci",
      "--default-authentication-plugin=mysql_native_password"
    ]
    networks:
      - appnet

  redis:
    image: redis:7-alpine
    container_name: laravel_redis
    ports:
      - "6379:6379"
    networks:
      - appnet

  mailpit:
    image: axllent/mailpit:latest
    container_name: mailpit
    ports:
      - "8025:8025"
      - "1025:1025"
    networks:
      - appnet

  node:
    build:
      context: ./docker/node
    container_name: nextjs_frontend
    working_dir: /usr/src/app
    volumes:
      - ./frontend:/usr/src/app
    ports:
      - "3000:3000"
    environment:
      - NEXT_PUBLIC_API_BASE=http://localhost:8080
      - TZ=Asia/Tokyo
    depends_on:
      - web
    networks:
      - appnet

volumes:
  dbdata:

networks:
  appnet:
    driver: bridge

```

## GitHub連携
・アカウント作成
・リポジトリ作成
・最初のコミット
・（ブランチ名の変更）
・最初のプッシュ

```
git remote add origin https://github.com/<ユーザー名>/<リポジトリ名>.git
```
remote：リモート（外部）のリポジトリを登録する
origin：リモートリポジトリの名前（慣習的に「origin」と呼ぶ）
URL：GitHub の自分のリポジトリのアドレス

確認コマンド
```
git remote -v

```
origin https://github.com/ユーザー名/リポジトリ名.git (fetch)
origin https://github.com/ユーザー名/リポジトリ名.git (push)
のように出れば OK。
